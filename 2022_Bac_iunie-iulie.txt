#include <iostream>

using namespace std;

int main()
{
   int a[101][101], n, m, s = 0;
   cin >> m >> n;
   for(int i = 1; i <= m; i++){
       for(int j = 1; j <= n; j++){
            cin >> a[i][j];
        }
   }
   for(int i = 1; i <= m; i++){
       for(int j = 1; j <= n; j++){
            int minn = 11;
            if(a[i-1][j] < minn && i - 1 > 0) minn = a[i - 1][j];
            if(a[i][j-1] < minn && j - 1 > 0) minn = a[i][j - 1];
            if(a[i][j+1] < minn && j + 1 <= n) minn = a[i][j + 1];
            if(a[i + 1][j] < minn && i + 1 <= m) minn = a[i + 1][j];
            if(a[i][j] < minn){
                s = s + (minn - a[i][j]);
            }
        }
   }
   cout << s;
}
