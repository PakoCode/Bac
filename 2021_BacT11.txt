void imog(int a, int b, int &rez){
    int x = 0, y = 0, p = 1;
    while(a != 0){
        if(a % 2 != 0){
            x = a % 10 * p + x;
            p = p * 10;
        }
        a = a / 10;
    }
      while(b != 0){
        if(b % 2 != 0){
            y = b % 10 + y * 10;
        }
        b = b / 10;

    if(x == y) rez = 1;
}

    imog(a, b, rez);
